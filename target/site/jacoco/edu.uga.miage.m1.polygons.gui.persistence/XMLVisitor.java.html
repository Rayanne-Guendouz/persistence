    @Override
    public void visit(ComplexShape complexShape) {
        appendRepresentation(String.format(START_COMPLEX));
        SimpleShapeBis[] shapes = complexShape.getShapes();
        for (SimpleShapeBis shape : shapes) {
            shape.accept(this);
        }
        appendRepresentation(END_COMPLEX);
    }
